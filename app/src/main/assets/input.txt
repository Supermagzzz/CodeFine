#include <vector>
#include <algorithm>

using namespace std;

struct Person
{
string name;
int score;
};

bool cmp(Person a, Person b)
{
return a.score > b.score;
}

int main()
{
	int n;
	cin >> n;
	vector<Person> v(n);
	for(int i = 0; i < n; i++)
	{
		cin >> v[i].name >> v[i].score;
	}
	sort(v.begin(), v.end(), cmp);
	for(int i = 0; i < n; i++)
	{
		cout << v[i].name << " " << v[i].score; << endl;
	}
}
