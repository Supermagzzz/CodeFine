C. Перепись

В переменной num мы будем хранить текущий номер самого взрослого мужчины, а в переменной age его возраст.

Решение задачи:

#include <iostream>

using namespace std;

int main()
{
int n;
cin >> n;
int num = -1, age = -1;
for(int i = 1; i <= n; i++)
{
int a, s;
cin >> a >> s;
if(s == 1 && a > age)
{
age = a;
num = i;
}
}
cout << num;
}
