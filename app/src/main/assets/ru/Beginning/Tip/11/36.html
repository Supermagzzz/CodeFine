E. Антон и Даник

Решение задачи:

#include <iostream>
#include <string>

using namespace std;
int main()
{
int n, cntA = 0, cntD = 0;
cin >> n;
string s;
cin >> s;
for(int i = 0; i < s.size(); i++)
{
if(s[i] == 'A')
{
cntA++;
}
else
{
cntD++;
}
}
if(cntA == cntD)
{
cout << "Friendship";
}
else if(cntA > cntD)
{
cout << "Anton";
}
else
{
cout << "Danik";
}
}
