B. Камни на столе

Разделим слово на блоки состоящие из одинаковых букв. Если превратить каждый такой блок в одну букву, то получится ответ. Значит, нужно посчитать количество лишних букв в этих блоках. Переменная last содержит символ, из которого состоит текущий блок.

Решение задачи:

#include <iostream>
#include <string>

using namespace std;
int main()
{
string s;
cin >> s;
char last = s[0];
int cnt = 0;
for(int i = 1; i < s.size(); i++)
{
if(s[i] == last)
{
cnt++;
}
else
{
last = s[i];
}
}
cout << cnt;
}
