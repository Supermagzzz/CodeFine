F. IQ тест

Решение задачи:

#include <iostream>

using namespace std;
int main()
{
int n, cntC = 0, cntN = 0, valC, valN;
cin >> n;
for(int i = 0; i < n; i++)
{
int x;
cin >> x;
if(x % 2 == 0)
{
cntC++;
valC = x;
}
else
{
cntN++;
valN = x;
}
}
if(cntC == 1)
{
cout << valC;
}
else
{
cout << valN;
}
}
