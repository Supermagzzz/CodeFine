A. Драконы

Очевидно, что драконов надо убивать в порядке увеличения силы. Так, дойдя до очередного дракона, мы наберем максимальную возможную силу. Осталось проверить, сможем ли мы убить всех.

Решение задачи:

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

struct Dragon
{
int x, y;
};

bool cmp(Dragon a, Dragon b)
{
return a.x < b.x;
}

int main()
{
int s, n;
cin >> s >> n;
vector<Dragon> v(n);
for(Dragon &e : v)
{
cin >> e.x >> e.y;
}
sort(v.begin(), v.end(), cmp);
for(int i = 0; i < n; i++)
{
if(v[i].x >= s)
{
cout << "NO";
return 0;
}
s += v[i].y;
}
cout << "YES";
}
