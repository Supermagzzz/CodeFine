B. Башни

Нужно найти количество разных чисел в массиве и число, которое встречается чаще всего. Давайте создадим массив, где индекс это число. Проходя по массиву из задачи будем прибавлять единицу в наш массив в нужный индекс.
Так мы получим массив, где для каждого числа написано его количество. Осталось посчитать, сколько ненулевых ячеек в таком массиве и найти максимальное число в нем.

Решение задачи:

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
int n;
cin >> n;
vector<int> v(1001);
for(int i = 0; i < n; i++)
{
int x;
cin >> x;
v[x]++;
}
int cnt = 0, maxV = 0;
for(int i : v)
{
if(i != 0)
{
cnt++;
}
if(i > maxV)
{
maxV = i;
}
}
cout << maxV << " " << cnt;
}
