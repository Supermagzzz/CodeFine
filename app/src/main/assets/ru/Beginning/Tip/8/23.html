C. Контроперация

Решение задачи:

#include <iostream>

using namespace std;

int main()
{
int n;
cin >> n;
int a[1000];
int maxV = 0, minV = 1001;
for(int i = 0; i < n; i++)
{
int x;
cin >> x;
if(maxV < x)
{
maxV = x;
}
if(minV > x)
{
minV = x;
}
}
for(int i = 0; i < n; i++)
{
if(a[i] == maxV)
{
a[i] = minV;
}
cout << a[i] << " ";
}
}
