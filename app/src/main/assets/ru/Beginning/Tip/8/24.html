Решение задачи:

#include <iostream>
#include <string>

using namespace std;

int main()
{
int n;
cin >> n;
string a[100];
for(int i = 0; i < n; i++)
{
cin >> a[i];
}
int xE, yE, xS, yS;
for(int i = 0; i < n; i++)
{
for(int j = 0; j < n; j++)
{
if(a[i][j] == 'E')
{
xE = j + 1;
yE = n - i;
}
if(a[i][j] == 'S')
{
xS = j + 1;
yS = n - i;
}
}
}
cout << xE << " " << yE << endl;
cout << xS << " " << yS << endl;
}
