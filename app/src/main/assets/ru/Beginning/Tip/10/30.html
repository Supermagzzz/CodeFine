A. Операции над массивом

Решение задачи:

#include <iostream>
#include <string>

using namespace std;


int a[1000];

int get(int i)
{
return a[i - 1];
}

void update(int l, int r, int x)
{
for(int i = l - 1; i <= r - 1; i++)
{
a[i] = x;
}
}

void add(int l, int r, int x)
{
for(int i = l - 1; i <= r - 1; i++)
{
a[i] += x;
}
}

int rsq(int l, int r)
{
int sum = 0;
for(int i = l - 1; i <= r - 1; i++)
{
sum += a[i];
}
return sum;
}

int rmq(int l, int r)
{
int minValue = 2000000000;
for(int i = l - 1; i <= r - 1; i++)
{
if(a[i] < minValue)
{
minValue = a[i];
}
}
return minValue;
}
