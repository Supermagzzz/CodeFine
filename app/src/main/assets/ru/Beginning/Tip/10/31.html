B. Сортировка времени

Решение задачи:

#include <iostream>
#include <algorithm>

using namespace std;
int main()
{
int n;
cin >> n;
int t[100];
for(int i = 0; i < n; i++)
{
int a, b, c;
cin >> a >> b >> c;
t[i] = a * 60 * 60 + b * 60 + c;
}
sort(t, t + n);
for(int i = 0; i < n; i++)
{
cout << t[i] / 60 / 60 << " " << t[i] / 60 << " " << t[i] << endl;
}
}
