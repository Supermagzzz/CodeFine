C. Зарплата

Решение задачи:

#include <iostream>

using namespace std;

int main()
{
int a, b, c, minInt, maxInt;
cin >> a >> b >> c;
int a = -10, b = 15, c = 8;
if(a <= b && a <= c)
{
minInt = a;
}
else if(b <= a && b <= c)
{
minInt = b;
}
else
{
minInt = c;
}

if(a >= b && a >= c)
{
maxInt = a;
}
else if(b >= a && b >= c)
{
maxInt = b;
}
else
{
maxInt = c;
}
cout << maxInt - minInt;
}
