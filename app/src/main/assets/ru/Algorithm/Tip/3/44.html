D. Настя и массив

Ответ равен количеству разных элементов в массиве. Если мы отсортируем массив, то разные элементы будут находиться рядом. Значит - количество разных элементов, это количество мест в массиве, где соседи не равны друг другу.

Решение задачи:

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
int n;
cin >> n;
vector<int> v(n);
for(int &e : v)
{
cin >> e;
}
sort(v.begin(), v.end());
int cnt = 1;
for(int i = 1; i < n; i++)
{
if(v[i] != v[i - 1])
{
cnt++;
}
}
cout << cnt;
}
