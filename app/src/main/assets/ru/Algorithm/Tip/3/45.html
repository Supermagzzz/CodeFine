E. Сортируем массив

Если массив был отсортирован и в нем был перевернут подотрезок, значит элементы в массиве будут сначала возрастать (не относящиеся к отрезку), потом убывать (раньше возрастали, но входят в отрезок, который их перевернул) и потом опять возрастать (не относящиеся к отрезку). Найдем места, где меняется возрастание/убывание и получим границы отрезка. Останется только проверить, что развернув его мы получим отсортированный массив. Если массив был отсортирован сразу, выведем 1 1, так как переворот одного элемента ничего не меняет.

Решение задачи:

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
int n;
cin >> n;
vector<int> v(n);
for(int &e : v)
{
cin >> e;
}
vector<int> sorted = v;
sort(sorted.begin(), sorted.end());
if(v == sorted)
{
cout << "yes" << endl;
cout << "1 1";
}
else
{
bool phase = 1;
int l = 0, r = n;
for(int i = 1; i < n; i++)
{
if(phase)
{
if(v[i - 1] > v[i])
{
l = i;
phase = 0;
}
}
else
{
if(v[i - 1] < v[i])
{
r = i;
break;
}
}
}
reverse(v.begin() + l, v.begin() + r + 1);
if(v == sorted)
{
cout << "yes" << endl;
cout << l << " " << r;
}
else
{
cout << "no" << endl;
}
}
}
