A. Игра

Первый игрок всегда будет удалять максимальное число (так как хочет минимизировать результат), а второй будет всегда удалять минимальное число. Значит, если мы отсортируем изначальный массив, то ответ на задачу будет находиться в середине.

Решение задачи:

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
int n;
cin >> n;
vector<int> v(n);
for(int &i : v)
{
cin >> i;
}
sort(v.begin(), v.end());
if(v.size() % 2 == 1)
{
cout << v[v.size() / 2];
}
else
{
cout << v[v.size() / 2 - 1];
}
}
