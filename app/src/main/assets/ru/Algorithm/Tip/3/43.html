C. Флеш-карты

Лучше всего сначала записывать на карты большего размера.

Решение задачи:

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
int n, m;
cin >> n >> m;
vector<int> v(n);
for(int &e : v)
{
cin >> e;
}
sort(v.begin(), v.end());
int cnt = 0;
int cur = n - 1;
while(m > 0)
{
m -= v[cur];
cur--;
cnt++;
}
cout << cnt;
}
